package banking;

public class Checking extends Account {
 double deposit(double amount) {
		
	    credit(amount);
		double a=getBalance();
		setBalance(a);
		return getBalance();
	}
	double withdraw(double amount) {
		
		
		if(amount>=50000) {
			
			deduct(amount);
		}
		else {
			debit(amount);
			
		}
		double a=getBalance();
		setBalance(a);
		return a;
		
		
	}
	
	double checkbalance() {
		
		return getBalance();
	}
	
}
