package banking;

public class Savings  extends Account{
	double deposit(double amount) {
		credit(amount);
		double a=getBalance()+(getBalance()*0.07);
		setBalance(a);
		return a;
		
	}
	
	double withdraw(double amount) {
		
		debit(amount);
		double a=getBalance();
		setBalance(a);
		return a;
		
	}
	
	double checkbalance() {
		
		return getBalance();
	}
	

	
	

}
